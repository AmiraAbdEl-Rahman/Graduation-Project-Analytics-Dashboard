# Group by month and team to calculate completed task percentage
df["task_created_month"] = df["task_created_at"].dt.to_period("M")

# Calculate completion rate per team per month
team_progress = df.groupby(["task_created_month", "team_id"])["task_status"].apply(lambda x: (x == "completed").sum() / len(x)).reset_index()

# Calculate the average progress of all teams
average_progress = team_progress.groupby("task_created_month")["task_status"].mean()

# Select a specific team for comparison 
selected_team_id = df["team_id"].mode()[0]  # Most frequent team
selected_team_progress = team_progress[team_progress["team_id"] == selected_team_id]

# Plot
plt.figure(figsize=(12, 6))
plt.plot(average_progress.index.astype(str), average_progress.values, label="Average Progress of All Teams", linestyle="--", color="gray")
plt.plot(selected_team_progress["task_created_month"].astype(str), selected_team_progress["task_status"], marker="o", label=f"Team {selected_team_id} Progress", color="blue")

plt.title(f"Progress of Team {selected_team_id} vs. Average of All Teams")
plt.xlabel("Month")
plt.ylabel("Task Completion Rate")
plt.legend()
plt.xticks(rotation=45)
plt.grid(True)
plt.show()
